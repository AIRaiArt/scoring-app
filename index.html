window.confirmResetRound = async function(roundId, roundName) {

  if (confirm(`รีเซ็ตคะแนนของ "${roundName}"?\nข้อมูลจะหายถาวร`)) {

    try {

      await remove(ref(db,`scores/${roundId}`));

      if (allScores[roundId]) delete allScores[roundId];

      renderDashboard();

      toast('รีเซ็ตแล้ว','#C0392B');

    } catch(e) { toast('ไม่สำเร็จ','#C0392B'); }

  }

};

window.addRound = function() {

  window.openModal(

    'เพิ่มรอบใหม่',

    'ตั้งชื่อรอบการแข่งขัน เช่น "รอบคัดเลือก", "รอบชิงชนะเลิศ"',

    '',

    async (name)=>{

      if (!name) return;

      const newRound = { id: 'r'+Date.now(), name, contestants: [] };

      rounds.push(newRound);

      renderSetup();

      toast('เพิ่มรอบแล้ว — อย่าลืมกด "บันทึก"');

    }

  );

};


window.deleteRound = function(roundId, roundName) {

  if (confirm(`ลบ "${roundName}"?\nคะแนนในรอบนี้จะถูกลบด้วย`)) {

    rounds = rounds.filter(r=>r.id!==roundId);

    renderSetup();

    toast('ลบรอบแล้ว — อย่าลืมกด "บันทึก"');

  }

};


window.editRoundContestants = function(roundId) {

  const round = rounds.find(r=>r.id===roundId);

  const contestants = round.contestants||[];

  const list = contestants.length ? contestants : Array.from({length:5},(_,i)=>({id:i+1,name:`ผู้เข้าแข่งขัน ${i+1}`}));


  window.openModal(

    `ผู้เข้าแข่งขัน — ${round.name}`,

    'แก้ไขชื่อผู้เข้าแข่งขัน (คั่นด้วยบรรทัดใหม่ 1 คนต่อบรรทัด)',

    list.map(c=>c.name).join('\n'),

    async (text)=>{

      const names = text.split('\n').map(n=>n.trim()).filter(Boolean);

      round.contestants = names.map((name,i)=>({id:i+1,name}));

      renderSetup();

      toast('อัปเดตรายชื่อแล้ว — อย่าลืมกด "บันทึก"');

    },

    true 

  );

};


window.saveSetup = async function() {

  rounds.forEach((r,i)=>{

    const el=document.getElementById(`round-name-${i}`); if(el) r.name=el.value.trim()||r.name;

  });

  judges.forEach((j,i)=>{

    const el=document.getElementById(`judge-name-${i}`); if(el) j.name=el.value.trim()||j.name;

  });

  criteria.forEach((cr,i)=>{

    const n=document.getElementById(`criteria-name-${i}`);

    const m=document.getElementById(`criteria-max-${i}`);

    if(n) cr.name=n.value.trim()||cr.name;

    if(m) cr.max=parseInt(m.value)||cr.max;

  });


  setSyncDot('setup-sync-dot','syncing');

  try {

    await update(ref(db,'/'), { judges, criteria, rounds });

    setSyncDot('setup-sync-dot','ok');

    toast('✓ บันทึกและ Sync แล้ว');

    renderSetup();

  } catch(e) {

    setSyncDot('setup-sync-dot','err');

    toast('❌ บันทึกไม่สำเร็จ','#C0392B');

  }

};


// ── MODAL (generic) ──

let modalTextarea = false;

window.openModal = function(title, sub, val, cb, multiline=false) {

  modalCallback = cb; modalTextarea = multiline;

  document.getElementById('modal-title').textContent = title;

  document.getElementById('modal-sub').textContent = sub;

  const inp = document.getElementById('modal-input');


  if (multiline) {

    const ta = document.createElement('textarea');

    ta.id='modal-input'; ta.className='modal-input'; ta.rows=8;

    ta.style.resize='vertical'; ta.value=val; ta.placeholder='...';

    inp.parentNode.replaceChild(ta, inp);

    setTimeout(()=>{ ta.focus(); }, 80);

  } else {

    if (inp.tagName==='TEXTAREA') {

      const ni=document.createElement('input');

      ni.type='text'; ni.id='modal-input'; ni.className='modal-input';

      ni.onkeydown=(e)=>{ if(e.key==='Enter') window.modalConfirm(); };

      inp.parentNode.replaceChild(ni, inp);

    }

    document.getElementById('modal-input').value = val;

    setTimeout(()=>{ document.getElementById('modal-input').focus(); document.getElementById('modal-input').select(); }, 80);

  }

  document.getElementById('modal').style.display='flex';

};


window.modalConfirm = function() {

  const val = document.getElementById('modal-input').value;

  window.closeModal();

  if (modalCallback) modalCallback(val.trim());

};


window.closeModal = function() {

  document.getElementById('modal').style.display='none';

  modalCallback=null;

};


window.onModalBackdrop = function(e) {

  if (e.target===document.getElementById('modal')) window.closeModal();

};


// ── HELPERS ──

function isScoredComplete(jId,rId,cId){

  const s=allScores[rId]?.[jId]?.[cId]||{};

  return criteria.every(cr=>s[cr.id]!==undefined);

}

function isJudgeRoundComplete(jId,rId){

  const contestants=(rounds.find(r=>r.id===rId)?.contestants)||[];

  return contestants.length>0 && contestants.every(c=>isScoredComplete(jId,rId,c.id));

}

function getScoredCount(jId,rId){

  const contestants=(rounds.find(r=>r.id===rId)?.contestants)||[];

  return contestants.filter(c=>isScoredComplete(jId,rId,c.id)).length;

}

function setSyncDot(id,state){const d=document.getElementById(id);if(d)d.className='sync-dot '+state;}

function toast(msg,bg){

  const t=document.getElementById('toast');

  t.textContent=msg; t.style.background=bg||'var(--green)';

  t.classList.add('show'); setTimeout(()=>t.classList.remove('show'),2500);

}
